update_good_production_removal = {
    if = {
        limit = {
            has_province_modifier = good_production_$VALUE$_$GOOD$
            FROM = { has_country_flag = good_production_$VALUE$_$GOOD$_boost }
            NOT = { owner = { has_country_flag = good_production_$VALUE$_$GOOD$_boost } }
        }
        remove_province_modifier = good_production_$VALUE$_$GOOD$
    }
}

update_good_production_addition = {
    if = {
        limit = {
            trade_goods = $GOOD$
            NOT = { has_province_modifier = good_production_$VALUE$_$GOOD$ }
            FROM = { has_country_flag = good_production_$VALUE$_$GOOD$_boost }
            NOT = { owner = { has_country_flag = good_production_$VALUE$_$GOOD$_boost } }
        }
        
        add_province_modifier = { 
            name = good_production_$VALUE$_$GOOD$
            duration = -1
            hidden = yes
        }
    }
}

add_effect_good_production = {
    custom_tooltip = good_production_$VALUE$_$GOOD$_tt
        
    hidden_effect = {
        set_country_flag = good_production_$VALUE$_$GOOD$_boost
        
        every_owned_province = {
            if = {
                limit = {
                    trade_goods = $GOOD$
                }
                add_province_modifier = { 
                    name = good_production_$VALUE$_$GOOD$
                    duration = -1
                    hidden = yes
                }
            }
        }
    }
}

remove_effect_good_production = {
    hidden_effect = {

    }

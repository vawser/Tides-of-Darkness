teleport_units_with_type_and_amount_effect = {
	if = {
		limit = {
			event_target:sender = {
				num_of_units_in_province = {
					who = ROOT
					amount = $amount$
					type = $unittype$
				}
			}
		}
		event_target:sender = {
			kill_units = {
				who = ROOT
				amount = $amount$
				type = $unittype$
			}
			clr_province_flag = sender_province
		}
		event_target:receiver = {
			add_unit_construction = {
				type = $unittype$
				amount = $amount$
				speed = 0.01
				cost = 0
			}
		}
	}
}

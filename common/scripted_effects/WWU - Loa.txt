# Loa
increase_loa_influence = {
    if = {
        limit = {
            has_country_modifier = loa_influence_high
        }
        custom_tooltip = LOA_INFLUENCE_REMAINS_HIGH
    }
    else_if = {
        limit = {
            has_country_modifier = loa_influence_medium
        }
        
        custom_tooltip = LOA_INFLUENCE_SET_TO_HIGH
        
        hidden_effect = {
            remove_country_modifier = loa_influence_medium
            add_ruler_modifier = {
                name = loa_influence_high
                duration = -1
            }
        }
    }
    else_if = {
        limit = {
            has_country_modifier = loa_influence_low
        }
        
        custom_tooltip = LOA_INFLUENCE_SET_TO_MEDIUM
        
        hidden_effect = {
            remove_country_modifier = loa_influence_low
            add_ruler_modifier = {
                name = loa_influence_medium
                duration = -1
            }
        }
    }
    else = {
        custom_tooltip = LOA_INFLUENCE_SET_TO_LOW
        
        hidden_effect = {
            add_ruler_modifier = {
                name = loa_influence_low
                duration = -1
            }
        }
    }
}

# Divine Fate
#-----------------------------------
# Grant Divine Fated nations a bonus during war to allow them to blob easier and to become a threat to the player.
# Bonus only applies in AI vs. AI wars.

divine_fate_early_game = {
	potential = {
        ai = yes
        has_country_modifier = divine_fate
    }
    trigger = {
        ai = yes
        has_country_modifier = divine_fate
        
        # Before 100 years has passed
        NOT = {
            had_global_flag = {
                flag = game_duration_timer
                days = 36500
            }
        }
    }
    
    # Bonuses
    core_creation = -0.1
    governing_capacity_modifier = 0.1
    max_absolutism = 10
    
    production_efficiency = 0.1
    global_tax_modifier = 0.1
    global_manpower_modifier = 0.1
    global_sailors_modifier = 0.1
    
    build_cost = -0.1
    build_time = -0.1
    
    idea_cost = -0.1
    
    ae_impact = -0.1
    
    legitimacy = 1
    republican_tradition = 1
    devotion = 1
    horde_unity = 1
    meritocracy = 1
    
    prestige = 1.0
    
    global_unrest = -2
    stability_cost_modifier = -0.1
    
    global_colonial_growth = 10
    
    diplomats = 1
    merchants = 1
    
    army_tradition = 0.5
    navy_tradition = 0.5
    
    land_forcelimit_modifier = 0.1
    naval_forcelimit_modifier = 0.1
    
    all_estate_loyalty_equilibrium = 0.05
}

divine_fate_ai_war_early_game = {
	potential = {
        ai = yes
        has_country_modifier = divine_fate
    }
    trigger = {
        # Before 100 years has passed
        NOT = {
            had_global_flag = {
                flag = game_duration_timer
                days = 36500
            }
        }
        
        # Not involved in a war with the player
        NOT = {
            any_country = {
                war_with = ROOT
                ai = no
            }
        }
        # Involved in a war with AI
        any_country = {
            war_with = ROOT
            ai = yes
        }
    }
    
    land_morale = 0.05
    discipline = 0.05
    siege_ability = 0.05
    movement_speed = 0.05
    
    reinforce_speed = 0.1
    manpower_recovery_speed = 0.1
    recover_army_morale_speed = 0.1
    
    land_attrition = -0.1
    war_exhaustion = -0.025
}

divine_fate_focus_eastern_kingdoms = {
	potential = {
        ai = yes
        has_country_modifier = divine_fate
        capital_scope = {
            continent = continent_eastern_kingdoms
        }
    }
    trigger = {
        continent_eastern_kingdoms = {
            type = all
            owned_by = ROOT
        }
    }
    
    # No bonus, this is for AI aggression
    
    ai_weight = {
		factor = 1000 # AI will aim to achieve this
	}
}
